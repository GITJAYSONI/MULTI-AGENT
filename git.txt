# 1. Initialize git in your project folder (only run this ONCE per project)
git init

# 2. Stage all the files so Git knows you want to track them
git add .

# 3. Save these files into Git history with a message
git commit -m "Initial commit"

# 4. Rename the default branch to 'main' (this is the standard modern naming convention)
git branch -M main

# Point your local project to the GitHub HTTPS URL
git remote add origin https://github.com/GITJAYSONI/MULTI-AGENT.git

# Upload your code to GitHub (The "-u" remembers your branch for next time)
git push -u origin main

# Point your local project to the GitHub SSH URL
git remote add origin git@github.com:GITJAYSONI/MULTI-AGENT.git

# Upload your code to GitHub
git push -u origin main

# 1. Check what files you have modified or added
git status

# 2. Stage all your modified/new files
git add .

# 3. Commit the changes with a descriptive message 
git commit -m "Added a new feature for the agent"

# 4. Upload your new commits to GitHub
git push
